SELECT column FROM table...
  WHERE column 
	= value
	IN (list of scalar values)
	BETWEEN val1 AND val2
	LIKE simple_wildcard
		% = any number of characters
		_ = exactly one character
	> (SELECT ...)
	is NULL
  ORDER BY column
